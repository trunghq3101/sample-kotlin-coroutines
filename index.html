Sample Kotlin Coroutines

A coroutine promises to return a value: Concurrency for single value - async await.

import kotlinx.coroutines.* import kotlin.system.* suspend fun main() = coroutineScope { val time = measureTimeMillis { val one = async { delay(1000L) 1 } val two = async { delay(1000L) 2 } println("The answer is ${one.await() + two.await()}") } println("Completed in $time ms") } import kotlinx.coroutines.* import kotlin.system.* suspend fun main() = coroutineScope { val time = measureTimeMillis { val one = doOneAsync(this) val two = doTwoAsync(this) println("The answer is ${one.await() + two.await()}") } println("Completed in $time ms") } fun doOneAsync(scope: CoroutineScope) = scope.async { delay(1000L) 1 } fun doTwoAsync(scope: CoroutineScope) = scope.async { delay(1000L) 2 }
