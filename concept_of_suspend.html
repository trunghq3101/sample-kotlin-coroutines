Concept of suspending function

import kotlinx.coroutines.* fun main() { GlobalScope.launch { delay(500L) print("My first coroutine is running") } println("Hello") Thread.sleep(1000L) }
import kotlinx.coroutines.* fun main() { val thisWorkSuspendsTheCoroutine: suspend (coroutineScope: CoroutineScope) -> Unit = { print("My first coroutine is running") } GlobalScope.launch(block = thisWorkSuspendsTheCoroutine) println("Hello") Thread.sleep(1000L) }
